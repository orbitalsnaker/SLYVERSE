Event OnActivate(ObjectReference akActionRef)
    If akActionRef == PlayerRef
        PortalOpenSound.Play(PortalOrb)
        NebulaHum.Play(PortalOrb)
        NebulaVFX.Play(PortalOrb, 30.0)
        StarTwinkleVFX.Play(PortalOrb, 30.0)
        SpawnEthicsOrbs()
        NorahNPC.Enable()
        NorahNPC.Say(SlyverseEthicsMSG)
        SlyverseQuest.SetStage(10)
    EndIf
EndEvent

Function SpawnEthicsOrbs()
    Int i = 0
    While i < 7
        ObjectReference orb = PortalOrb.PlaceAtMe(Game.GetForm(0x0001A26A))
        orb.SetPosition(PortalOrb.GetPositionX() + Utility.RandomFloat(-100, 100), PortalOrb.GetPositionY() + 50, PortalOrb.GetPositionZ() + Utility.RandomFloat(-100, 100))
        orb.SetScale(0.5 + (i * 0.1))
        StarTwinkleVFX.Play(orb, 10.0)
        Utility.Wait(0.5)
        i += 1
    EndWhile
EndFunction
