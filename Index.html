const mateEternal = () => {
  if (localStorage.getItem('mateEternal') !== 'unlocked') { document.getElementById('paywall').style.display='flex'; return; }
  speak("MATE ETERNAL ∞");
  if (canvas.requestFullscreen) canvas.requestFullscreen();

  const vs = `attribute vec2 a;uniform float t;uniform vec2 p;void main(){vec2 pos=a+vec2(sin(t*0.2),cos(t*0.3))*0.1;gl_Position=vec4(pos,0,1);gl_PointSize=5.0;}`;
  const fs = `uniform float t;void main(){vec2 uv=gl_PointCoord-0.5;float d=length(uv);float glow=0.15/(d+0.02);float h=fract(t*0.05+d*8.0);vec3 c=0.5+0.5*cos(6.28*(h+vec3(0.0,0.33,0.67)));gl_FragColor=vec4(c*glow,1.0);}`;
  program = gl.createProgram();
  [gl.VERTEX_SHADER, gl.FRAGMENT_SHADER].forEach((type,i)=>{
    const s=gl.createShader(type);
    gl.shaderSource(s, type===gl.VERTEX_SHADER?vs:fs);
    gl.compileShader(s);
    gl.attachShader(program,s);
  });
  gl.linkProgram(program);

  const buffer = gl.createBuffer();
  gl.bindBuffer(gl.ARRAY_BUFFER, buffer);
  const base = new Float32Array(131000*2);
  for(let i=0,j=0;i<131000;i++){
    const a=i*0.00012,r=0.1+Math.sin(a*13)*0.3;
    base[j++]=Math.cos(a)*r;base[j++]=Math.sin(a)*r;
  }
  gl.bufferData(gl.ARRAY_BUFFER, base, gl.STATIC_DRAW);

  currentGame = () => {
    t += 0.016;
    gl.clearColor(0,0,0,1); gl.clear(gl.COLOR_BUFFER_BIT);
    gl.useProgram(program);
    gl.uniform1f(gl.getUniformLocation(program,'t'),t);
    gl.enableVertexAttribArray(0);
    gl.vertexAttribPointer(0,2,gl.FLOAT,false,0,0);
    gl.drawArrays(gl.POINTS,0,131000);
    if(t>50){overlay.textContent="GRACIAS POR EXISTIR";overlay.style.opacity=1;}
  };
};
